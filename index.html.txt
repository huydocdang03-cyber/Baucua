<script src="https://cdn.socket.io/4.5.4/socket.io.min.js"></script>
<script>
const socket = io("https://YOUR_SERVER_LINK");

function join(){
    const name = document.getElementById("name").value;
    const role = document.getElementById("role").value;

    socket.emit("join", {name, role});
    document.getElementById("game").style.display="block";
}

function bet(item){
    socket.emit("bet", {item, amount:100});
}

function roll(){
    socket.emit("roll");
}

function generate(){
    socket.emit("admin_generate");
}

socket.on("preview", r=>{
    alert("Admin thấy trước: "+r.join("-"));
});

socket.on("dealer", d=>{
    document.getElementById("dealer").innerText="Dealer: "+d;
});

socket.on("users", u=>{
    let me = u[Object.keys(u).find(k=>k===socket.id)];
    if(me){
        document.getElementById("money").innerText="Tiền: "+me.money;
    }
});

socket.on("result", r=>{
    document.getElementById("result").innerText="KQ: "+r.join("-");
});
</script>
